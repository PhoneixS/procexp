def startproperties_25process():
    #process explorer full size
    type(Key.SPACE, KEY_ALT)
    type("x")

    click(Pattern().targetOffset(-40,14))

    doubleClick()

    wait(, 20)

def test_cpu25percent():
    """test a cpu bound process to take 25% CPU at a 4 core machine"""
